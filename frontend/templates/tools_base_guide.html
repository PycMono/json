JavaScript 
// Decimal to other bases
(255).toString(2);    // "11111111"
(255).toString(8);    // "377"
(255).toString(16);   // "ff"

// Parse from any base
parseInt("ff", 16);   // 255
parseInt("1010", 2);  // 10

// BigInt for large numbers

big.toString(2);
Python 
# Decimal to other bases
bin(255)   # '0b11111111'
oct(255)   # '0o377'
hex(255)   # '0xff'

# Parse from any base
int("ff", 16)    # 255
int("1010", 2)   # 10

# Format without prefix
format(255, 'b')   # '11111111'
format(255, 'x')   # 'ff'

Go 
import "strconv"

// Decimal to other bases
